public class Main {
    public static void main(String[] args) {

        // Összegzés - Egy tömb összes elemének összeadása
        int[] tombOsszeg = {3, 8, 2, 4, 5, 1, 6};
        int osszeg = 0;
        for (int index = 0; index < tombOsszeg.length; index++) {
            osszeg += tombOsszeg[index]; // Az aktuális elem hozzáadása az összeghez
        }
        System.out.println(osszeg); // Az összeg kiíratása

        // Megszámolás - Hány elem nagyobb, mint 5?
        int[] tombSzamlalas = {3, 8, 2, 4, 5, 1, 6};
        int szamlalo = 0;
        for (int index = 0; index < tombSzamlalas.length; index++) {
            if (tombSzamlalas[index] > 5) { // Ha az adott elem nagyobb, mint 5
                szamlalo++; // Növeljük a számlálót
            }
        }
        System.out.println(szamlalo); // Az eredmény kiíratása

        // Eldöntés tétel - Megnézzük, hogy van-e a tömbben egy adott szám
        int[] tombDontes = {3, 8, 2, 4, 5, 1, 6};
        int keresettSzam = 2;
        int index = 0;
        while (index < tombDontes.length && tombDontes[index] != keresettSzam) {
            index++; // Ha nem találjuk, lépünk a következő elemre
        }
        if (index < tombDontes.length) {
            System.out.println("Van ilyen szám."); // Ha megtaláltuk
        } else {
            System.out.println("Nincs ilyen szám."); // Ha nincs benne
        }

        // Kiválasztás tétel - Megkeressük egy adott elem indexét
        int[] tombKivalasztas = {3, 8, 2, 4, 5, 1, 6};
        int keresettIndex = 2;
        int pozicio = 0;
        while (tombKivalasztas[pozicio] != keresettIndex) {
            pozicio++; // Lépünk a következő elemre
        }
        System.out.printf("%d\n", pozicio + 1); // Az index kiíratása (1-től kezdődő sorszámmal)

        // Maximum kiválasztás - Legnagyobb elem megkeresése
        int[] tombMax = {3, 8, 2, 4, 5, 1, 6};
        int maxErtek = tombMax[0];
        for (int i = 0; i < tombMax.length; i++) {
            if (tombMax[i] > maxErtek) {
                maxErtek = tombMax[i]; // Új maximum érték beállítása
            }
        }

        System.out.println("Legnagyobb: " + maxErtek); // Legnagyobb szám kiíratása

        // Minimum kiválasztás - Legkisebb elem megkeresése
        int[] tombMin = {3, 8, 2, 4, 5, 1, 6};
        int minErtek = tombMin[0];
        for (int i = 0; i < tombMin.length; i++) {
            if (tombMin[i] < minErtek) {
                minErtek = tombMin[i]; // Új minimum érték beállítása
            }
        }

        System.out.println("Legkisebb: " + minErtek); // Legkisebb szám kiíratása

    }
}
