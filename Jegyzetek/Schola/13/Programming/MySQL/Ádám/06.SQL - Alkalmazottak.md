## Lekérdezés
```sql
-- Lekérdezünk mindent
SELECT * from alkalmazottak;
-- 1. Id es teljes név, ahol a vetezéknév 1. betűje az 'h'
SELECT alk_id vezetlid, veznev, kernev 
FROM alkalmazottak 
WHERE veznev 
LIKE '_h%' ORDER BY veznev;
-- 2. Teljes név és felvétel dátuma, akiket 1997-benvagy 1998-ban vettek fel
SELECT veznev, kernev, felfevtel 
FROM alkalmazottak 
WHERE YEAR(felvetel) = 1997 OR YEAR(felvetel) = 1998;
-- 3. Vezetéknév és részleg kell, akiket 97-ben, vagy 98-ban vettek fel és a 3-as részlegben dolgoznak,
WHERE veznev, kernev, reszleg_id 
FROM alkalmazottak 
WHERE reszleg_id = 3 AND (YEAR(felvetel) = 1997 OR YEAR(felvetel) = 1998)
-- 4. Id-k és fizetések, kellenek, ahol az utobbik az nem 2500 és 2900 között van, csökken.
SELECT alk_id, fizu
FROM alkalmazottak
WHERE fizu NOT BETWEEN 2500 AND 2900
ORDER BY fizu DESC;
-- 5. Összes Id érték, ahol az allas_id nem 7, 8 vagy 9. (nem használható: <, >, ND, OR).
SELECT alk_id
FROM alkalmazottak
WHERE állas_id NOT IN (7, 8, 9);
-- 6. Azon rekordok, a nélkülhol a vezetéknév vége az 'are' + még egy betű
SELECT *
FROM alkalmazottak
WHERE veznev LIKE '%are_';
-- 7. Azon rekordok, ahol a vezetéknév eleje 'Jo', aztán még két betű van.
SELECT *
FROM alkalmazottak
WHERE veznev LIKE 'Jo__';
-- 8. Azon rekordok,  ahol a vezetéknév 'S'-el kezdődő. de nem 'Sh...'.
SELECT *
FROM alkalmazottak
WHERE veznev LIKE 'S%' AND veznev NOT LIKE 'Sh%';
-- 9. Mennyi az átlagfizetés (duplitákan fizetésértékek).
	-- Extra ehhez: az eredményt kerekítsük 2 tizedesre
SELECT ROUND(AVG(DISTINCT fizu), 2) AS atlagfizu
FROM alkalmazottak;
-- 10.Hányan dolgoznak a 6-s részlegen?
SELECT COUNT(*)
FROM alkalmazottak
WHERE reszleg_id = 6;
-- 11. Részlegenként jelenjenek meg az átlagfizetés.
SELECT reszleg_id, ROUND(AVG(fizu), 2) AS atlagfizu
FROM alkalmazottak
GROUP BY reszleg_id;
-- 12. Az a a rekord kell, akinek a legnagyobb a fozetése (LIMIT nem lehet).
SELECT *
FROM alkalmazottak
WHERE fizu = (SELECT MAX(fizu) FROM alkalmazottak);
-- 13. Teljes nevek kellenek, meg egy 'Új fizetés' oszlop az eredeti 110%-ával csökkenő sorban.
SELECT CONCAT(veznev, ' ', kernev) AS teljes_nev, 
       ROUND(fizu * 1.10, 2) AS uj_fizetes
FROM alkalmazottak
ORDER BY uj_fizetes DESC;
```
