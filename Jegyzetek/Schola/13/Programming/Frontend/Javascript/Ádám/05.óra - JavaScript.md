function szórendez(szo) {
  let betutomb = szo.split("");//A szóból csinál egy betűtömböt
  let nagybetűk = [];
  let kisbetűk = [];
  //Végignézzük a tömb betűit:
  for (let i = 0; i < betutomb.length; i++){
    //Ha az adott betű egyezik önmaga nagybetűs változatával, akkor eleve nagybetűs volt, így abba a tömbbe kerül bele:
    if (betutomb[i] == betutomb[i].toUpperCase()) {
      nagybetűk.push(betutomb[i]);
    } else {//Egyéb esetben kisbetűnek kellett lennie:
      kisbetűk.push(betutomb[i]);
    }
  }
  //Úgy kérem vissza, hogy a két tömböt megfelelő sorrendben egyberakom és össze is tolom egy string-be:
  return nagybetűk.concat(kisbetűk).join("");
}

function doboztérfogat(...dobozok){//Határozatlan elemszám jelölése: "..."
  let össztérfogat = 0; //Ez lesz a megoldás
  //Közben a gép az beteszi a sok 3 elemű listát egy "dobozok" nagylistába
  for (let i = 0; i<dobozok.legth; i++){ //itt egy db "i" az 1 doboz
    let egydoboz = dobozok[i];  //Tartalom: a soron következő 3 elemű lista
    let térfogat = 1; //Ez egy db doboz térfogata lesz.
    for (let j = 0; j < egydoboz.length; j++){
      térfogat = térfogat * egydoboz[j];//térfogat megszorzása a doboz 3 kiterjedésével
    }
      összérfogat += térfogat; //adott doboz térfogatának hozzáadása az össztérfogathoz
  }
  return össztérfogat;
}
