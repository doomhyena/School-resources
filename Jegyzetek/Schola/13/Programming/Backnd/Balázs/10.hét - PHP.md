## 1. hetfo/reg.php

```php
<?php 
	session_start();
	
	// Regisztráció
	if(isset($_POST['reg-btn'])){
		$vane = false;
		
		// Vizsgálat, hogy regisztrálták-e már a felhasználónevet
		for($i=0;$i<count($_SESSION['felhasznalok']);$i++){
			if($_SESSION['felhasznalok'][$i]['felhasznalonev'] == $_POST['username']){
				$vane = true;
			}
		}
		
		// Találtunk-e egyező felhasználót
		if($vane == true){
			echo "Már létezik ilyen felhasználó!";
		}
		else{
			$uj_felhasznalo = array();
			$uj_felhasznalo['felhasznalonev'] = $_POST['username'];
			$uj_felhasznalo['jelszo'] = $_POST['password'];
			array_push($_SESSION['felhasznalok'], $uj_felhasznalo);
			header("Location: login.php");
		}
	}
?>
```

## 3. hetfo/login.php

```php
<?php 
	session_start();
	
	// Bejelentkezés
	if(isset($_POST['login-btn'])){
		$vane = false;
		
		for($i=0;$i<count($_SESSION['felhasznalok']);$i++){
			if($_SESSION['felhasznalok'][$i]['felhasznalonev'] == $_POST['username']){
				$vane = true;
			}
		}
		
		if($vane == true){
			for($i=0;$i<count($_SESSION['felhasznalok']);$i++){
				if($_SESSION['felhasznalok'][$i]['jelszo'] == $_POST['password'] && $_SESSION['felhasznalok'][$i]['felhasznalonev'] == $_POST['username']){
					$_SESSION['felhasznalo'] = $_POST['username'];
					header("Location: index.php");
				}
			}
		}
		else{
			echo "Nincs ilyen felhasználó!";
		}
	}
?>
```

## 4. pentek/reg.php

```php
<?php 
	session_start();
	
	// Regisztráció
	if(isset($_POST['reg-btn'])){
		$vane = false;
		
		// Megvizsgáljuk, hogy létezik-e már fiók azzal a felhasználónévvel
		for($i=0;$i<count($_SESSION['felhasznalok']);$i++){
			if($_SESSION['felhasznalok'][$i]['felhasznalonev'] == $_POST['username']){
				$vane = true;
			}
		}
		
		// Ha nem -> regisztrálás
		if($vane == false){
			// Megvizsgáljuk, hogy a két beírt jelszó egyezik-e
			if($_POST['password1'] == $_POST['password2']){
				$uj_felhasznalo = array();
				$uj_felhasznalo['felhasznalonev'] = $_POST['username'];
				$uj_felhasznalo['jelszo'] = $_POST['password1'];
				array_push($_SESSION['felhasznalok'], $uj_felhasznalo);
				header("Location: login.php");
			}
			else{
				echo "A két jelszó nem egyezik!";
			}
		}
		else{
			echo "Már létezik ilyen fiók!";
		}
	}
?>

<h1>Regisztráció</h1>
<form method="post">
	<input type="text" name="username" placeholder="Felhasználónév">
	<br><br>
	<input type="text" name="password1" placeholder="Jelszó">
	<br><br>
	<input type="text" name="password2" placeholder="Jelszó újra">
	<br><br>
	<input type="submit" name="reg-btn" value="Regisztráció">
</form>
```

## 6. pentek/index.php

```php
<?php 
	session_start();
	
	// Felhasználók tárolása
	if(!isset($_SESSION['felhasznalok'])){
		$_SESSION['felhasznalok'] = array();
	}
	
	// Van-e bejelentkezett felhasználó

		header("Location: reg.php");

		$uj_poszt['iro'] = $_SESSION['felhasznalo'];

	}
?>
```
