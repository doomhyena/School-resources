### Form1.Designer.cs

- Meghatározza az alkalmazás felhasználói felületét (UI), beleértve a haladási sávot és a gombokat (pl. `Start`, `Stop`).
- A haladási sáv egyedi (`ProgBarPlus`), és olyan jellemzőkkel rendelkezik, mint a színátmenet, a fokozatos színes háttér, és a szöveges kijelzés.
- Két gombkészlet található: az egyik a háttérfolyamatot vezérli (`btn_Start`, `btn_Stop`), a másik az `async/await` módszert (`btn_Start2`, `btn_Stop2`) használja.

### Form1.cs

- Itt van implementálva a gombokhoz tartozó logika.
- **BackgroundWorker**:
    - A `btn_Start` gomb megnyomásakor elindul a `BackgroundWorker`, amely a háttérben frissíti a haladási sávot.
    - A háttérfolyamat egy ciklust futtat, amely időnként frissíti a haladási sávot a `ReportProgress` használatával, miközben figyeli a megszakítást (`backgroundWorker1.CancellationPending`).
    - A haladási sáv frissítése a `backgroundWorker1_ProgressChanged` metóduson keresztül történik.
    - A folyamat befejezésekor, vagy ha megszakítják, egy üzenetablak jelenik meg.
    - A `btn_Stop` gomb leállítja a háttérfolyamatot.
- **Async/Await és CancellationTokenSource**:
    - A `btn_Start2` gomb elindítja a progresszív sávot egy aszinkron módszerrel (`AsyncAwaitProgress`), amely a `Task.Run` segítségével végzi a munkát, és a `Progress<int>` segítségével jelenti a haladást.
    - A haladási sáv frissítése a `progress.Report(i)` hívással történik.
    - A `cts.Cancel()` segítségével megszakítható a művelet, és a `btn_Stop2` gomb kezdeményezi a megszakítást.

### MyProgressBar.cs

- Egy egyedi progresszív sávot definiál (`MyProgressBar`), amely lehetővé teszi a kettős pufferelést, csökkentve a villogást.

### Program.cs

- Az alkalmazás belépési pontja, ahol a `Form1` ablak inicializálódik és elindul.

### Összegzés:

- Két módszert használsz a progresszív sáv frissítésére: az egyik a `BackgroundWorker`-t alkalmazza háttérfolyamatokhoz és haladási információk jelentéséhez, a másik az `async/await`-ot aszinkron feladatokhoz, amelyek támogatják a megszakítást.
- Mindkét módszer a saját gombjaival van összekötve (`Start`/`Stop`), és mindkettő lehetőséget ad a művelet közbeni leállításra.

```cs
﻿using System;
using System.ComponentModel;
using System.Threading;
using System.Threading.Tasks;
using System.Windows.Forms;

namespace ProgressBar
{
	public partial class Form1 : Form
	{
		CancellationTokenSource cts;
		public Form1()
		{
			InitializeComponent();

		}

		#region BackgroundWorker
		private void btn_Start_Click(object sender, EventArgs e)
		{
			backgroundWorker1.RunWorkerAsync();
		}

		private void TheTask()
		{
			for (int i = 0; i <= 100; i++)
			{
				System.Threading.Thread.Sleep(1000);
				progressBar1.Value = i;
			}
		}

		private void backgroundWorker1_DoWork(object sender, DoWorkEventArgs e)
		{
			for (int i = 0; i <= 100; i++)
			{
				if (backgroundWorker1.CancellationPending)
				{
					e.Cancel = true;
				}
				else
				{
					System.Threading.Thread.Sleep(100);
					backgroundWorker1.ReportProgress(i);
				}
			}
		}

		private void backgroundWorker1_ProgressChanged(object sender, ProgressChangedEventArgs e)
		{
			progressBar1.Value = e.ProgressPercentage;
		}

		private void backgroundWorker1_RunWorkerCompleted(object sender, RunWorkerCompletedEventArgs e)
		{
			if (e.Cancelled)
			{
				MessageBox.Show("Cancelled!");
			}
			else
			{
				MessageBox.Show("Done!");
			}

		}

		private void btn_Stop_Click(object sender, EventArgs e)
		{
			backgroundWorker1.CancelAsync();
		}
		#endregion

		#region Async - Await

		#endregion

		private async void btn_Start2_Click(object sender, EventArgs e)
		{
			cts = new CancellationTokenSource();
			progressBar1.Value = 0;
			Progress<int> progress = new Progress<int>(
				percent => progressBar1.Value = percent
				);
			await AsyncAwaitProgress(cts, progress);
		}

		private async Task AsyncAwaitProgress(CancellationTokenSource cts, IProgress<int> progress)
		{
			await Task.Run(() =>
			{
				for (int i = 0; i <= 100; i++)
				{
					Thread.Sleep(100);
					if (progress != null)
					{
						progress.Report(i);
					}
					if (cts.IsCancellationRequested)
					{
						break;
					}
				}
			});
		}

		private void btn_Stop2_Click(object sender, EventArgs e)
		{
			cts.Cancel();
		}
	}
}
```
