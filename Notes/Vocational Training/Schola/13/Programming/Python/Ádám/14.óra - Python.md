def időbe(perc):
    nap = perc // 1440
    ora = (perc%1440) // 60
    perc = perc % 60
    return nap, ora, perc
#fájlbeolvasás:
adatok = []
f = open("lista.txt")
i = 1
egyfilm = ""
for adat in f.readlines():
    egyfilm += adat
    if i % 5 == 0:
        ido, cim, evad_hossz, hossz, latta = egyfilm.strip().split("\n") #soronként tagoljuk
        adatok.append([ido,cim,evad_hossz,int(hossz),int(latta)])
        egyfilm = ""
    i += 1
f.close()

#6. feladat - A hét mely napja esik az adott dátumra:
def hetnapja(ev,ho, nap):
    ev4 = ev // 4
    ev100 = ev//100
    ev400 = 400
    napok = ["v","h","k","sze","cs","p","szo"]
    honapok = [0,3,2,5,0,3,5,1,4,6,2,4]
    if ho < 3:
        ev = ev-1
    return napok[(ev+ev4-ev100+ev400+honapok[ho-1]+nap)%7]
#------------------------
