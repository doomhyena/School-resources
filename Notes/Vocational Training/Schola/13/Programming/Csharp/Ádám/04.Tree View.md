﻿using System;
using System.Collections.Generic;
using System.ComponentModel;
using System.Data;
using System.Drawing;
using System.Linq;
using System.Text;
using System.Threading.Tasks;
using System.Windows.Forms;

namespace TreeView
{
	public partial class Form1 : Form
	{
		List<Category> categories = new List<Category>()
		{
			new Category() { ID=1,Name="Monitor" },
			new Category() { ID=2,Name="Motherboard"},
			new Category() { ID=3,Name="CPU"}
		};
		List<Product> products = new List<Product>()
		{
			new Product(){ID=1,Name="Samsung LCD",Category=1 },
			new Product(){ID=2,Name="ASUS kjfhgkjddfb",Category=2 },
			new Product(){ID=3,Name="Intel i7-8888",Category=3 },
		};
		public Form1()
		{
			InitializeComponent();
		}

		private void Form1_Load(object sender, EventArgs e)
		{
			TreeNode root = new TreeNode("Products");
			//treeView1.Nodes.Add(root);
			TreeNode tn, tn_l1;

			foreach (Category cat in categories)
			{
				tn = new TreeNode(cat.Name) { Name = cat.ID.ToString() };
				foreach (Product prod in products.Where(q => q.Category == cat.ID))
				{
					tn_l1 = new TreeNode(prod.Name) { Name = prod.ID.ToString() };
					tn.Nodes.Add(tn_l1);
				}
				root.Nodes.Add(tn);
			}
			treeView1.Nodes.Add(root);
		}
	}

	public class Category
	{
		public int ID { get; set; }
		public string Name { get; set; }

	}

	public class Product
	{
		public int ID { get; set; }
		public string Name { get; set; }
		public int Category { get; set; }
	}
}
