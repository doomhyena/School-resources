```sql
CREATE TABLE ugynokok (
    ukod VARCHAR(255) PRIMARY KEY,
    unev VARCHAR(255),
    munkaterulet VARCHAR(255),
    megbizas DECIMAL(10, 2),
    mobil VARCHAR(255),
    orszag VARCHAR(255)
);
```
