/*
A függvény kiveszi a duplikátumokat és növekvő sorba rendezi a listát.
Pl.: ([1, 4, 4, 4, 4, 4, 3, 2, 1, 2]) ➞ [1, 2, 3, 4]
*/
function egyedirendez(lista) {
  //Itt az eredeti számok lesznek sorba rendezve:
  let rendezett = lista.sort(function(a,b) {return a-b});
  //Karaktergyűjtemény készítése (amiben minden előforduló elerm csak egyszer szerepel):
  return Array.from(new Set(rendezett));
  //Array: mert tömbből (array-ból) csináljuk
  //new Set(): legyen egy új Set (gyűjtemény) (a rendezett lista alapján)
}

/*
Kis- és nagybetűssé alakítás:
var betűa = 'A';
var betűb = 'b';
betűa.toLowerCase() --> kisbetűssé alakítás ('a')
betűb.toUpperCase() --> nagybetűssé alakítás ('B')

Megvizsgálunk egy stringet, amiben feltételezzük, hogy csak 'x' és 'o' betűk szerepelnek.
Ugyanannyi x és o betű van-e (true/false válasz). A kis-nagy betű különbsége nem számít.
Pl.: XO('oooxxo') --> false
     XO('OoXX') --> true
*/
function XO(szöveg) {
  var x = 0, o = 0; //ezek számolják majd, hogy melyikből hány darabot találunk
  for (var i = 0; i < szöveg.length; i++) { //végigmegy a szövegen
    if (szöveg[i].toLowerCase() === 'x') { //ha az adott karakter egy 'x'
      x += 1; //akkor az 'x' számlálója növekszik
    } else if (szöveg[i].toLowerCase() === 'o') { //ha az adott karakter egy 'o'
    o += 1; //akkor az 'o' számlálója növekszik
    }
  }
  return x === o; //a vizsgálat eredményét adja vissza
}

  szöveg.split('').forEach(function(c)){ //ahol 'c' = az épp vizsgált karakter

A függvény megvizsgál 1 db betűt, és megmondja, hogy milyen messze van a legközelebbi magánhangzótól. Pl.: lkmh('b') --> 'a'
  - Az angol abc szerint nézzük
  - Ha magánhangzót adunk meg, akkor értelemszerűen 0 a megoldás
  - Ha egyforma távolságra van 2 mh-tól, akkor az abc-ben hamarabbi lesz a megoldás

function lkmh(betű) {
  //Egy lista a magánhangzóknak:
  let mhlista = ['a','e','i','o','u'];
  //Külön eset, ha eleve egy magánhangzót adtunk meg:
  if (mhlista.includes(betű)) return betű; //listanév.includes(KeresettElem)
  //Mássalhangzó megadása esetén megkeressük, hogy hány betű eltérés van:
  let UjLista = mhlista.map(elem => Math.abs(elem.charCodeAt()-betű.charCodeAt(0)));
  //A talált különbségek közül ki kell keresni, hogy melyik a legkisebb szám és hogy az melyik betűhöz tartozik:
  return mhlista[UjLista.indexOf(Math.min(...UjLista))];
  //indexOf(): index alapján keresek meg egy listaelemet
  //...: határozatlan elemszám jelölése
  //Math.min(halmaz): a legkisebb szám keresése egy halmazban
}

/*
Egy futóval lépegetünk a sakktáblán. Megadunk egy kezdőpontot, egy végpontot és egy maximális lépésszámot. A kérdés, hogy eljuthatunk-e A-ból B-be annyi lépés alatt?
Pl.:
futó('a1','b4', 2) --> true
futó('a1','b5', 5) --> false //mert sehogy sem tud másik színre lépni a futó
*/
function futó(kezd,cél,n) {
  //Kiszámoljuk, hogy vízszintesen és függőlegesen is hány egység a különbség a 2 hely között:
    //(A betűs részeket karakterkóddá alakítom):
  xkülönbség = Math.abs(cél[0].charCodeAt(0)-kezd[0].charCodeAt(0));
    //(A szám résznél csak megbizonyosodok, hogy int típus-e):
  ykülönbség = Math.abs(parseInt(cél[1])-parseInt(kezd[1]));
  if ((xkülönbség+ykülönbség)%2 == 1) return false; //ha a 2 pont eltérő színre esik
  //Ha nem akadt fent a színeltérésen, akkor továbbmegy a következő if-re:
  if (n >= 2) return true; //Ha van legalább 2 lépésünk
  //Ha kettőnél kevesebb lépés van, akkor megnézzük, hogy esetleg egyezik-e a kezdő és a cél hely:
  if (kezd == cél) return true; //hiszen 0 lépés kell
  if (n == 1) { //Ha pontosan 1 lépésünk van rá
    if (xkülönbség != ykülönbség) return false;//Ha nem ugyanannyit megyünk mindkét irányba, akkor false
    else return true;
  }
  return false;
}
