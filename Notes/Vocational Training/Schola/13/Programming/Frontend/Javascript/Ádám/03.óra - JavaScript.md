Hány alkalommal volt nagyobb a szám az előzőnél?
Pl.: [10,11,12,9,10] --> 3

function nagyobbszám(számok){
  var eredmény = 0;
  for (var i = 1;i < számok.length;i++){//1-től a számlista hosszáig nő
    if (számok[i] > számok[i-1]){ //Ha az aktuális elem nagyobb az előzőnél
      eredmény++;//Akkor eredmény nő 1-el
    }
  }
  return eredmény;
}

console.log(nagyobbszám([10,11,12,9,10]))

A "vált" kifejezés megfordítja az igaz/hamis láncot.
Pl.: ["aev";"vált";"zkf";"vált";"adf"]; --> [true,false,false,true,true]

function vált(szavak){
  let megoldás = [];//ebbe kerülnek majd a true/false értékek
  let állapot = true;
  for (let i = 0; i<szavak.length; i++){
    if (szavak[i] === "vált"){//Ha azt találja, hogy "vált" (i. index)
        állapot = !állapot;//állapot változó az ellentétére változik
    }
    megoldás.push(állapot);
  }
  return megoldás;
}
