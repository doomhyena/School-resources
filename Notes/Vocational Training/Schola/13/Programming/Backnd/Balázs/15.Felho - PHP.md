## 3. /login.php

```php 
<?php 
	require "config.php";
	session_start();
	
	if(isset($_POST['login-btn'])){
		$username = $_POST['username'];
		$password = $_POST['password'];
		
		$lekerdezes = "SELECT * FROM users WHERE username='$username'";
		$talalt_felhasznalo = $conn->query($lekerdezes);
		
		if(mysqli_num_rows($talalt_felhasznalo) == 1){
			$felhasznalo = $talalt_felhasznalo->fetch_assoc();
			if($password == $felhasznalo['password']){
				$_SESSION['id'] = $felhasznalo['id'];
				header("Location: index.php");
			}
			else{
				echo "Helytelen jelszó!";
			}
		}
		else{
			echo "Nincs ilyen felhasználó!";
		}
	}
?>
<h1>Bejelentkezés</h1>
<form method="post">
	<input type="text" name="username" placeholder="Felhasználónév">
	<br><br>
	<input type="password" name="password" placeholder="Jelszó">

	<input type="submit" name="login-btn" value="Belépek!">
</form>
```

## 4. /reg.php

```php 
<?php 
	require "config.php";
	
	if(isset($_POST['reg-btn'])){
		$username = $_POST['username'];
		$password = $_POST['password'];
		
		$lekerdezes = "SELECT * FROM users WHERE username='$username'";
		$talalt_felhasznalo = $conn->query($lekerdezes);
		
		if(mysqli_num_rows($talalt_felhasznalo) == 0){
			$conn->query("INSERT INTO users VALUES(id, '$username', '$password')");
			header("Location: login.php");
		}
		else{
			echo "Már van ilyen nevű felhasználó!";
		}
	}
?>
<h1>Regisztráció</h1>
```
