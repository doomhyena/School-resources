## 1. bejelentkezes.php

```php
<?php 
	session_start();
	
	if(isset($_POST['gomb'])){
		for($i=0;$i<count($_SESSION['szovegek']);$i++){
			if($_SESSION['szovegek'][$i]['felhasznalonev'] == $_POST['username']){
				if($_SESSION['szovegek'][$i]['jelszo'] == $_POST['password']){
					header("Location: index.php");
				}
				else{
					echo "Helytelen jelszó!";
				}
			}
		}
	}
?>

</form>
```
