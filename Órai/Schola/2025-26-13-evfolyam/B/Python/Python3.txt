'''
Ciklusok: egy x alkalommal megismérlődő művelet
Példa: Hány darab "a" betű van a "program" szóban?
    Megoldási mód:
        - a ciklus végigmegy a szó összes betűjén
        - betűnként megnézi, hogy az épp "a" betű-e 
        - Ha igen, akkor azt feljegyzi magának
'''
szoveg = "program"
db = 0 #Darab (talált "a" betűk száma lesz)
for karakter in szoveg: #A ciklus menjen végig a szöveg változón
    #for = ciklus parancs; karakter = aktuálisan nézett elem
    if karakter == "a": #"Ha az aktuális elem egy "a" betű"
        db += 1 #"Akkor bd nő 1-el"
print(db)

#Több ágú feltételek: több if ág van az else előtt.
#Példa: Vizsgáljuk meg a "Minta Elemér" nevet. Melyik része a hosszabb?
név = "Minta Elemér"
#Tagoljuk fel a nevet (szóköz alapján) és tároljuk el 2 változóban:
vezeteknev = név.split()[0]
keresztnev = név.split()[1]
#Vizsgálat elvégzése:
if len(vezeteknev) > len(keresztnev): #Ha a vezetéknév áll több karakterből
    print("A vezetéknév a hosszabb.")
elif len(vezeteknev) < len(keresztnev): #Ha a keresztnév áll több karakterből
    print("A keresztnév a hosszabb.")
else: #Egyéb esetben (ami csak az lehet, hogy egyenlő hosszúak)...
    print("Egyenlő hosszúak.")
