#A lista mely szavában van a legtöbb magánhangzó?
szavak = ["piros", "sárga", "zöld", "kék", "narancs"]
mhlista = "aáeéiíoóöőuúüű"
legtobb = szavak[0]
max = 0
for szo in szavak:
    db = 0
    for betu in szo:
        if betu in mhlista:
            db += 1
    if db > max:
        max = db
        legtobb = szo
print(legtobb)

# 1. Hozz létre egy számlistát és írd ki, hányszor fordul elő a 7-es szám!
lista = [3, 7, 2, 9, 4, 7, 6]
db = 0
for szam in lista:
    if szam == 7:
        db += 1
print(db)

# 2. Készíts egy másik számlistát. Aztán hozz létre egy új listát, amelyben minden szám fele van benne!
szamok = [10, 20, 30, 40, 50]
felek = []
for szam in szamok:
    felek.append(szam / 2)
print(felek)

# 3. Ez a kód egy string lista legrövidebb elemét keresse meg!
allatok = ["kutya", "macska", "kígyó", "hal", "oroszlán"]
rovid = allatok[0]
for allat in allatok:
    if len(allat) < len(rovid):
        rovid = allat
print(rovid)

# 4. Ez a kód adja össze egy számlista páratlan számait egy összegbe!
szamok = [5, 8, 11, 14, 17]
osszeg = 0
for szam in szamok:
    if szam % 2 != 0:
        osszeg += szam
print(osszeg)

# 5. Egy számlistában cseréld fel egymással a legkisebb és legnagyobb számok pozícióját!
szamok = [1, 4, 9, 16, 25]
maximumIndex = 0
minimumIndex = 0
for i in range(len(szamok)):
    if szamok[i] > szamok[maximumIndex]:
        maximumIndex = i
    if szamok[i] < szamok[minimumIndex]:
        minimumIndex = i
seged = szamok[maximumIndex]
szamok[maximumIndex] = szamok[minimumIndex]
szamok[minimumIndex] = seged
print(szamok)
